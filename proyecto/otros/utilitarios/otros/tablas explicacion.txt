 pruba del query para traer los pagos de las compra que como empresa tengo.

select c.codigo as compra,ct.nombre as tipo_de_compra,c.ncf,c.cod_suplidor,t.nombre as suplidor,c.rnc,c.monto,(select sum(monto) from factura_suplidor_pagos fs where fs.codigo_factura=c.codigo) as pagado from compra c join suplidor s
on c.cod_suplidor=s.codigo join compra_vs_producto cp on c.codigo=cp.cod_compra join tercero t on t.codigo=s.codigo join compra_tipo ct
on ct.codigo=c.cod_tipo where c.estado='1'
